Build and install everything automatically (basic package set)
	./auto-build.sh
  OR
	./auto-build-debug.sh

Fetch package sources from official sites
	./get.sh [PACKAGE_NAME [latest]]

Build packages
	./build.sh [PACKAGE_NAME]
  OR
	./build-debug.sh [PACKAGE_NAME]

Install packages
	./install.sh [PACKAGE_NAME]
